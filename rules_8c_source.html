311 static lsdn_err_t flush_fl_rule(struct lsdn_flower_rule *fl, struct lsdn_ruleset_prio *prio, bool update)
312 {
313  struct lsdn_ruleset *ruleset = prio->parent;
314  struct lsdn_filter *filter = lsdn_filter_flower_init(
315  ruleset->iface->ifindex, fl->fl_handle, ruleset->parent_handle,
316  ruleset->chain, prio->prio + ruleset->prio_start);
317  if (!filter)
318  return LSDNE_NOMEM;
319 
320  if (update)
321  lsdn_filter_set_update(filter);
322 
323  lsdn_log(LSDNL_RULES, "fl_%s(handle=0x%x)\n", update ? "update" : "create", fl->fl_handle);
324 
325  uint16_t ethtype;
326  if (!find_common_ethtype(fl, prio, &ethtype))
327  // TODO: also do the validation on VR layer and report the problem correctly there
328  abort();
329 
330  if (ethtype != ETH_P_ALL)
331  lsdn_flower_set_eth_type(filter, htons(ethtype));
332 
333  for(int i = 0; i<LSDN_MAX_MATCHES; i++) {
334  union lsdn_matchdata *match = &fl->matches[i];
335  union lsdn_matchdata *mask = &prio->masks[i];
336  switch(prio->targets[i]) {
337  case LSDN_MATCH_DST_MAC:
338  lsdn_flower_set_dst_mac(filter, match->mac.chr, mask->mac.chr);
339  break;
340  case LSDN_MATCH_SRC_MAC:
341  lsdn_flower_set_src_mac(filter, match->mac.chr, mask->mac.chr);
342  break;
343  case LSDN_MATCH_SRC_IPV4:
344  lsdn_flower_set_src_ipv4(filter, match->ipv4.chr, mask->ipv4.chr);
345  break;
346  case LSDN_MATCH_DST_IPV4:
347  lsdn_flower_set_dst_ipv4(filter, match->ipv4.chr, mask->ipv4.chr);
348  break;
349  case LSDN_MATCH_SRC_IPV6:
350  lsdn_flower_set_src_ipv6(filter, match->ipv6.chr, mask->ipv6.chr);
351  break;
352  case LSDN_MATCH_DST_IPV6:
353  lsdn_flower_set_dst_ipv6(filter, match->ipv6.chr, mask->ipv6.chr);
354  break;
355  case LSDN_MATCH_ENC_KEY_ID:
356  lsdn_flower_set_enc_key_id(filter, fl->matches[i].enc_key_id);
357  break;
358  case LSDN_MATCH_NONE:
359  break;
360  default:
361  abort();
362  }
363  }
364 
365  lsdn_flower_actions_start(filter);
366  size_t order = 1;
367  lsdn_foreach(fl->sources_list, sources_entry, struct lsdn_rule, r) {
368  assert (order + r->action.actions_count <= LSDN_MAX_ACT_PRIO);
369  r->action.fn(filter, order, r->action.user);
370  order += r->action.actions_count;
371  }
372 
373  lsdn_flower_actions_end(filter);
374  lsdn_err_t err = lsdn_filter_create(ruleset->ctx->nlsock, filter);
375  if (err != LSDNE_OK) {
376  lsdn_filter_free(filter);
377  return err;
378  }
379 
380  lsdn_filter_free(filter);
381 
382  return LSDNE_OK;
383 }

431 void lsdn_rule_apply_mask(
432  struct lsdn_rule *r, enum lsdn_rule_target targets[], union lsdn_matchdata masks[])
433 {
434  for(int i = 0; i<LSDN_MAX_MATCHES; i++) {
435  if (!lsdn_target_supports_masking(targets[i])) {
436  switch (targets[i]) {

448  r->matches[i].bytes[j] &= masks[i].bytes[j];

454 lsdn_err_t lsdn_ruleset_add(struct lsdn_ruleset_prio *prio, struct lsdn_rule *rule)
455 {
456  bool update = true;
457  lsdn_err_t err = LSDNE_OK;
458  rule->prio = prio;
459  rule->ruleset = prio->parent;
460  lsdn_rule_apply_mask(rule, prio->targets, prio->masks);
461  lsdn_log(LSDNL_RULES, "ruleset_add(iface=%s, chain=%d, prio=0x%x)\n",
462  rule->ruleset->iface->ifname, rule->ruleset->chain, prio->prio);
463  dump_rule(rule);
464 
465  struct lsdn_flower_rule *fl;
466  HASH_FIND(hh, rule->prio->hash_fl_rules, rule->matches, sizeof(rule->matches), fl);
467  if (!fl) {
468  uint32_t handle;
469  if (!lsdn_idalloc_get(&rule->prio->handle_alloc, &handle))
470  return LSDNE_NOMEM;
471 
472  fl = malloc(sizeof(*fl));
473  if (!fl) {
474  lsdn_idalloc_return(&rule->prio->handle_alloc, handle);
475  return LSDNE_NOMEM;
476  }
477  memcpy(fl->matches, rule->matches, sizeof(fl->matches));
478  lsdn_list_init(&fl->sources_list);
479  fl->fl_handle = handle;
480  HASH_ADD(hh, rule->prio->hash_fl_rules, matches, sizeof(fl->matches), fl);
481  update = false;
482  }
483  rule->fl_rule = fl;
484 
485  /* insert sorted into the sources list */
486  struct lsdn_list_entry *nearest_lower = &fl->sources_list;
487  lsdn_foreach(fl->sources_list, sources_entry, struct lsdn_rule, r) {
488  if (r->subprio == rule->subprio) {
489  /* there is no need to free `fl`, we won't be here if it is newly allocated */
490  return LSDNE_DUPLICATE;
491  }
492  if (r->subprio > rule->subprio)
493  break;
494  nearest_lower = &r->sources_entry;
495  }
496  lsdn_list_init_add(nearest_lower, &rule->sources_entry);
497 
498  err = flush_fl_rule(fl, prio, update);
499  if (err != LSDNE_OK) {
500  lsdn_list_remove(&rule->sources_entry);
501  acc_inconsistent(&err, free_fl_rule(fl, prio));
502  }
503  return err;
504 }

515 static bool lsdn_find_free_action(
516  struct lsdn_broadcast *br, size_t actions,
517  struct lsdn_broadcast_filter** out_filter, size_t* out_action_index)
518 {

522  struct lsdn_broadcast_action **a = &f->actions[i];
523  if(!*a) {

544  for(size_t i = 0; i<LSDN_MAX_ACT_PRIO - 1; i++) {

556 static lsdn_err_t lsdn_flush_action_list(struct lsdn_broadcast_filter* br_filter)
557 {
558  struct lsdn_broadcast *br = br_filter->broadcast;
559  struct lsdn_filter *filter = lsdn_filter_flower_init(
560  br->iface->ifindex,
561  MAIN_RULE_HANDLE, LSDN_INGRESS_HANDLE, br->chain, br_filter->prio);
562  lsdn_filter_set_update(filter);
563  size_t order = 1;
564 
565  lsdn_flower_actions_start(filter);
566  for(size_t i = 0; i < LSDN_MAX_ACT_PRIO - 1; i++) {
567  struct lsdn_broadcast_action *action = br_filter->actions[i];
568  if (!action)
569  continue;
570  //printf("Adding action to filter at %d if %s\n", order, iface->ifname);
