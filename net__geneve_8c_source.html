10 static lsdn_err_t geneve_use_stunnel(struct lsdn_phys_attachment *a)
11 {
12  lsdn_err_t err = LSDNE_OK;
13  struct lsdn_settings *s = a->net->settings;
14  struct lsdn_context *ctx = s->ctx;
15  struct lsdn_if *tunnel = &s->geneve.tunnel;
16  struct lsdn_ruleset *rules_in = &s->geneve.ruleset_in;
17  if (s->geneve.refcount == 0) {
18  err = lsdn_link_geneve_create(
19  ctx->nlsock, tunnel, lsdn_mk_iface_name(ctx), s->geneve.port,
20  true, ctx->overwrite);
21  if (err != LSDNE_OK) {
22  return err;
23  }
24 
25  err = lsdn_link_set(ctx->nlsock, tunnel->ifindex, true);
26  if (err != LSDNE_OK) {
27  acc_inconsistent(&err, lsdn_link_delete(ctx->nlsock, tunnel));
28  return err;
29  }
30 
31  err = lsdn_prepare_rulesets(ctx, tunnel, rules_in, NULL);
32  if (err != LSDNE_OK) {
33  acc_inconsistent(&err, lsdn_link_delete(ctx->nlsock, tunnel));
34  return err;
35  }
36 
37  err = lsdn_sbridge_phys_if_init(
38  ctx, &s->geneve.tunnel_sbridge, tunnel, true, rules_in);
39  if (err != LSDNE_OK) {
40  acc_inconsistent(&err, lsdn_link_delete(ctx->nlsock, tunnel));
41  return err;
42  }
43  }
44  s->geneve.refcount++;
45 
46  return err;
47 }
