902 static void validate_rules(struct lsdn_virt *virt, struct vr_prio *ht_prio)
903 {
904  struct vr_prio *prio, *tmp;
905  HASH_ITER(hh, ht_prio, prio, tmp) {
906  /* First check that matches are compatible */
907  struct lsdn_vr *first_rule = NULL;
908  lsdn_foreach(prio->rules_list, rules_entry, struct lsdn_vr, r) {
909  if (first_rule) {
910  bool same =
911  (memcmp(first_rule->targets, r->targets, sizeof(r->targets)) == 0)
912  && (memcmp(first_rule->masks, r->masks, sizeof(r->masks)) == 0);
913  if (!same) {
914  lsdn_problem_report(
915  virt->network->ctx, LSDNP_VR_INCOMPATIBLE_MATCH,
916  LSDNS_VR, first_rule,
917  LSDNS_VR, r,
918  LSDNS_VIRT, virt,
919  LSDNS_END);
920  /* do not bother doing more checks */
921  return;
922  }
923  } else {
924  first_rule = r;
925  }
926  }
927 
928  /* Optional todo:re use the hash table for commited rules that already exists.
929  * This will reduce memory usage (we can reuse the hh from the rule) and we can
930  * get the running time proportional to changed rules, not all rules. */
931 
932  /* Then check for conflicting rules */
933  struct lsdn_vr *rule_ht = NULL;
934  lsdn_foreach(prio->rules_list, rules_entry, struct lsdn_vr, r) {
935  struct lsdn_vr *duplicate;
936  lsdn_rule_apply_mask(&r->rule, r->targets, r->masks);
937  HASH_FIND(hh, rule_ht, r->rule.matches, sizeof(r->rule.matches), duplicate);
938  if (duplicate) {
939  lsdn_problem_report(
940  virt->network->ctx, LSDNP_VR_DUPLICATE_RULE,
941  LSDNS_VR, r,
942  LSDNS_VR, duplicate,
943  LSDNS_VIRT, virt,
944  LSDNS_END);
945  HASH_CLEAR(hh, rule_ht);
946  return;
947  }
948  HASH_ADD(hh, rule_ht, rule.matches, sizeof(r->rule.matches), r);
949  }
950  HASH_CLEAR(hh, rule_ht);
951  }
952 }

954 static void commit_vr(
955  struct lsdn_virt *virt, struct vr_prio *prio,
956  struct lsdn_vr *vr, enum lsdn_direction dir)
957 {
958  if (prio->commited_count == 0) {
959  assert(!prio->commited_prio);
960  /* This is not reversed: the egress from the virt is our ingress and vice versa */
961  struct lsdn_ruleset *rs = (dir == LSDN_IN ? &virt->rules_out : &virt->rules_in);
962  vr->rule.subprio = LSDN_VR_SUBPRIO;
963  prio->commited_prio = lsdn_ruleset_define_prio(rs, prio->prio_num);
964  if (!prio->commited_prio)
965  abort();
966  memcpy(prio->commited_prio->targets, vr->targets, sizeof(vr->targets));
967  memcpy(prio->commited_prio->masks, vr->masks, sizeof(vr->masks));
968  }
969 
970  lsdn_err_t err = lsdn_ruleset_add(prio->commited_prio, &vr->rule);
971  if (err != LSDNE_OK)
972  abort();
973  prio->commited_count++;
974 }

1574 lsdn_err_t lsdn_commit(struct lsdn_context *ctx, lsdn_problem_cb cb, void *user)
1575 {
1576  /* Error handling strategy:
1577  * If commit/decommit functions return void, it is their responsibility to report the problem and
1578  * mark the object as inconsistent as needed. Typically, the will use mark_commit_err for that.
1579  *
1580  * If the functions return lsdn_err_t, the caller has the responsibility to report the problem.
1581  * Commit functions can return any errors (typically LSDNE_NETLINK, LSDNE_INCONSISTENT or LSDNE_NOMEM),
1582  * while decommit functions are allowed to only return LSDNE_INCONSISTENT.
1583  */
1584  trigger_startup_hooks(ctx);
1585 
1586  lsdn_err_t err = lsdn_validate(ctx, cb, user);
1587  if(err != LSDNE_OK)
1588  return err;
1589 
1590  err = lsdn_context_ensure_socket(ctx);
1591  if(err != LSDNE_OK) {
1592  lsdn_problem_report(ctx, LSDNP_NO_NLSOCK, LSDNS_END);
1593  return err;
1594  }
1595 
1596  /* List of objects to process:
1597  * setting, network, phys, physical attachment, virt
1598  * Settings, networks and attachments do not need to be committed in any way, but we must keep them
1599  * alive until PAs and virts are deleted. */
1600 
1601  /********* Decommit phase **********/
1602  lsdn_foreach(ctx->networks_list, networks_entry, struct lsdn_net, n) {
1603  lsdn_foreach(n->virt_list, virt_entry, struct lsdn_virt, v) {
1604  if (ack_decommit(&v->state)) {
1605  decommit_virt(v);
1606  ack_delete(v, virt_do_free);
1607  }
1608  }
1609  lsdn_foreach(n->attached_list, attached_entry, struct lsdn_phys_attachment, pa) {
1610  if (ack_decommit(&pa->state)) {
1611  decommit_pa(pa);
1612  ack_delete(pa, pa_do_free);
1613  }
1614  }
1615  if (ack_decommit(&n->state))
1616  ack_delete(n, net_do_free);
1617  }
1618 
1619  lsdn_foreach(ctx->phys_list, phys_entry, struct lsdn_phys, p){
1620  if (ack_decommit(&p->state))
1621  ack_delete(p, phys_do_free);
1622  }
1623 
1624  lsdn_foreach(ctx->settings_list, settings_entry, struct lsdn_settings, s) {
1625  if (ack_decommit(&s->state))
1626  ack_delete(s, settings_do_free);
1627  }
1628 
1629  /********* (Re)commit phase **********/
1630  /* first create physical attachments for local physes and populate
1631  * them with virts, remote PAs and remote virts */
1632  lsdn_foreach(ctx->phys_list, phys_entry, struct lsdn_phys, p){
1633  if (p->is_local) {
1634  p->committed_as_local = p->is_local;
1635  lsdn_foreach(
1636  p->attached_to_list, attached_to_entry,
1637  struct lsdn_phys_attachment, pa)
1638  {
1639  commit_pa(pa);
1640  }
1641  }
1642  }
1643 
1644  /********* Ack phase **********/
1645  lsdn_foreach(ctx->settings_list, settings_entry, struct lsdn_settings, s) {
1646  ack_state(&s->state);
1647  }
1648 
1649  lsdn_foreach(ctx->phys_list, phys_entry, struct lsdn_phys, p){
1650  ack_state(&p->state);
1651  }
1652 
1653  lsdn_foreach(ctx->networks_list, networks_entry, struct lsdn_net, n){
1654  ack_state(&n->state);
1655 
1656  lsdn_foreach(n->attached_list, attached_entry, struct lsdn_phys_attachment, pa) {
1657  ack_state(&pa->state);
1658  }
1659 
1660  lsdn_foreach(n->virt_list, virt_entry, struct lsdn_virt, v) {
1661  ack_state(&v->state);
1662  }
1663  }
1664 
1665  if (ctx->inconsistent)
1666  return LSDNE_INCONSISTENT;
1667  else if (ctx->problem_count > 0)
1668  return LSDNE_COMMIT;
1669  else
1670  return LSDNE_OK;
1671 }
