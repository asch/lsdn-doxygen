91 struct mnl_socket *lsdn_socket_init()
92 {
93  int err;
94  struct mnl_socket *sock = mnl_socket_open(NETLINK_ROUTE);
95  if (!sock)
96  return NULL;
97 
98  err = mnl_socket_bind(sock, 0, MNL_SOCKET_AUTOPID);
99  if (err) {
100  mnl_socket_close(sock);
101  return NULL;
102  }
103 
104  /* This requests extended errorr messages (an error text). */
105  int yes = 1;
106  err = mnl_socket_setsockopt(sock, NETLINK_EXT_ACK, &yes, sizeof(yes));
107  if (err) {
108  lsdn_log(LSDN_NLERR, "Extended messages not available, %d\n", err);
109  }
110 
111  /* This asks the kernel not to send back messages in ACKs, just headers */
112  err = mnl_socket_setsockopt(sock, NETLINK_CAP_ACK, &yes, sizeof(yes));
113  if (err) {
114  lsdn_log(LSDN_NLERR, "CAP_ACK not available %d\n", err);
115  }
116 
117  return sock;
118 }

133 static lsdn_err_t process_response(struct nlmsghdr *nlh, bool ignore_err)
134 {
135  struct nlmsgerr *resp;
136  lsdn_err_t ret;
137 
138  if (nlh->nlmsg_type == NLMSG_ERROR) {
139  resp = mnl_nlmsg_get_payload(nlh);
140 
141  if (resp->error) {
142  const char* msg = NULL;
143  struct nlattr *attr;
144  uint32_t flags = NLM_F_CAPPED | NLM_F_ACK_TLVS;
145  bool has_extack = (nlh->nlmsg_flags & flags) == flags;
146  if (has_extack) {
147  mnl_attr_for_each(attr, nlh, sizeof(*resp)) {
148  uint16_t type = mnl_attr_get_type(attr);
149  if (type == NLMSGERR_ATTR_MSG) {
150  if (mnl_attr_validate(attr, MNL_TYPE_STRING) < 0)
151  break;
152  msg = mnl_attr_get_str(attr);
153  }
154  }
155  }
156  if (!ignore_err)
157  report_nl_error(resp->error, msg);
158  ret = LSDNE_NETLINK;
159  } else {
160  ret = LSDNE_OK;
161  }
162  } else {
163  ret = LSDNE_NETLINK;
164  }
165 
166 
167  return ret;
168 }

170 static lsdn_err_t send_await_response(
171  struct mnl_socket *sock, struct nlmsghdr *nlh, bool ignore_err)
172 {
173  int ret;

176  if (ret == -1)
177  return LSDNE_NETLINK;

913 static double tick_per_sec = 0;

915 static void tc_init_fail(void)
916 {
917  fprintf(stderr, "tc timer initialization failed\n");
918  abort();
919 }
920 void tc_core_init(void)
921 {
922  /* Code taken nearly verbatim from TC */
923  FILE *fp;
924  uint32_t clock_res;
925  uint32_t t2us;
926  uint32_t us2t;
927 
928  fp = fopen("/proc/net/psched", "r");
929  if (fp == NULL)
930  tc_init_fail();
931 
932  if (fscanf(fp, "%08x%08x%08x", &t2us, &us2t, &clock_res) != 3)
933  tc_init_fail();
934  fclose(fp);
935 
936  /* compatibility hack: for old iproute binaries (ignoring
937  * the kernel clock resolution) the kernel advertises a
938  * tick multiplier of 1000 in case of nano-second resolution,
939  * which really is 1. */
940  if (clock_res == 1000000000)
941  t2us = us2t;
942 
943  double clock_factor = (double)clock_res;
944  tick_per_sec = (double)t2us / us2t * clock_factor;
945 }
