510  Tcl_IncrRefCount(obj);
511  int parts;

513  r = TCL_ERROR;

515  }
516

520  assert (r == TCL_OK);

522  if (ls_err != LSDNE_OK) {

524  goto err;
525  }
526 
527  if (parts == 2) {
528  Tcl_Obj *mask_obj;
529  r = Tcl_ListObjIndex(interp, obj, 0, &mask_obj);
530  assert (r == TCL_OK);
531  ls_err = lsdn_parse_mac(mask, Tcl_GetString(mask_obj));
532  if (ls_err != LSDNE_OK) {
533  r = tcl_error(interp, "can not parse mac address mask");
534  goto err;
535  }
536  } else {
537  *mask = lsdn_single_mac_mask;
538  }
539  } else {

549  struct tcl_ctx *ctx, Tcl_Interp *interp, Tcl_Obj *obj,
550  lsdn_ip_t *value, lsdn_ip_t* mask)
551 {
552  int r = TCL_OK;
553  obj = split_slashes(interp, ctx, obj);
554  Tcl_IncrRefCount(obj);
555  int parts;
556  if (Tcl_ListObjLength(interp, obj, &parts) != TCL_OK) {
557  r = TCL_ERROR;
558  goto err;
559  }
560 
561  if (parts == 1 || parts == 2) {
562  Tcl_Obj *value_obj;
563  r = Tcl_ListObjIndex(interp, obj, 0, &value_obj);
564  assert (r == TCL_OK);
565  lsdn_err_t ls_err = lsdn_parse_ip(value, Tcl_GetString(value_obj));
566  if (ls_err != LSDNE_OK) {
567  r = tcl_error(interp, "can not parse mac address");
568  goto err;
569  }
570 
571  if (parts == 2) {
572  Tcl_Obj *mask_obj;
573  int prefix;
574  r = Tcl_ListObjIndex(interp, obj, 1, &mask_obj);
575  assert (r == TCL_OK);
576  r = Tcl_GetIntFromObj(interp, mask_obj, &prefix);
577  if (r == TCL_OK) {
578  /* the second part is network prefix */
579  if (!lsdn_is_prefix_valid(value->v, prefix)) {
580  r = tcl_error(interp, "the network prefix has invalid size for given IP version");
581  goto err;
582  }
583  *mask = lsdn_ip_mask_from_prefix(value->v, prefix);
584  } else {
585  r = TCL_OK;
586  /* the second part is mask or malformed */
587  ls_err = lsdn_parse_ip(mask, Tcl_GetString(mask_obj));
588  if (ls_err != LSDNE_OK) {
589  r = tcl_error(interp, "can not parse mac address mask");
590  goto err;
591  }
592  }
593  } else {
594  if (value->v == LSDN_IPv4) {
595  *mask = lsdn_single_ipv4_mask;
596  } else {
597  *mask = lsdn_single_ipv6_mask;
598  }
599  }
600  } else {
601  r = tcl_error(interp,
602  "Mac match has too many parts. Expected either a single ip, "
603  "ip-value/ip-mask pair or ip-value/prefix pair.");
604  goto err;
605  }
606 
607  if (value->v != mask->v)
608  r = tcl_error(interp,"The IP protocol versions for the value and mask do not match");
609 err:
610  Tcl_DecrRefCount(obj);

729 static int parse_units(Tcl_Interp *interp, const char *value, float *out, const struct unit *unit_tab)
730 {
731  /* Pre-validate the floating point format to (\d+)(.\d+)?<unit>, stricter than strtod */
732  const char *c = value;
733  /* We assume C locale here */
734  if(!isdigit(*c))
735  return tcl_error(interp, "numeric value must start with a digit");
736  while(isdigit(*c))
737  c++;
738  if (*c == '.') {
739  c++;
740  if(!isdigit(*c))
741  return tcl_error(interp, "digit must follow after '.'");
742  while(isdigit(*c))
743  c++;
744  }
745  const char *unit_start = c;
746  while(isalpha(*c))
747  c++;
748  if (*c)
749  return tcl_error(interp, "garbage foun after numeric value");

753  break;
754  }
755  if (!unit_tab->name)
756  return tcl_error(interp, "unknown unit");
757  *out = strtof(value, NULL) * unit_tab->scale;
